	\subsection*{1. Экспонента} \[ e^x = \sum_{n=0}^{\infty} \frac{x^n}{n!} \quad \text{на } (-\infty, \infty), \quad \text{так как} \] 
	
	\textbf{а)} частичные суммы ряда Тейлора \( S_n(x) = T_n(x) \), где \[ T_n(x) = \sum_{k=0}^{n} \frac{f^{(k)}(0)}{k!} x^k \] — многочлен Тейлора; \textbf{б)} \( \forall x \in (-\infty, \infty) \quad \exists H > 0 : x \in [-H, H] \), \[ |(e^x)^{(n)}| = |e^x| \leq e^H \quad \forall x \in [-H, H], \quad \text{следовательно,} \quad e^x = \sum_{n=0}^{\infty} \frac{x^n}{n!} \quad \text{на } [-H, H], \] \[ \text{следовательно,} \quad e^x = \sum_{n=0}^{\infty} \frac{x^n}{n!} \quad \forall x \in (-\infty, \infty). \]

	\subsection*{2. Косинус} \[ \cos x = \sum_{n=0}^{\infty} (-1)^n \frac{x^{2n}}{(2n)!} \quad \text{на } (-\infty, \infty), \] \text{так как} 
	
	\textbf{а)} частичные суммы ряда Тейлора \( S_n(x) = T_n(x) \), где \[ T_n(x) = \sum_{k=0}^{n} \frac{f^{(k)}(0)}{k!} x^k \quad \text{– многочлен Тейлора;} \] \textbf{б)} \[ \forall x \in (-\infty, \infty) \quad \exists H > 0 : x \in [-H, H], \] \[ |(\cos x)^{(n)}| = |\cos(x + \frac{\pi}{2}n)| \leq 1 \quad \forall x \in [-H, H], \quad \text{следовательно,} \] \[ \cos x = \sum_{n=0}^{\infty} (-1)^n \frac{x^{2n}}{(2n)!} \quad \text{на } [-H, H], \quad \text{следовательно,} \] \[ \cos x = \sum_{n=0}^{\infty} (-1)^n \frac{x^{2n}}{(2n)!} \quad \forall x \in (-\infty, \infty). \]

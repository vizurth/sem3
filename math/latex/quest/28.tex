{
	\section{Абсолютно сходящиеся ряды.}

\subsection*{Теорема 3.1.} Если сходится ряд \( \sum_{n=1}^{\infty} |a_n| \), то сходится и \( \sum_{n=1}^{\infty} a_n \).

\subsection*{Определение 3.1.}
\begin{itemize}
	\item Если сходится ряд \(\sum_{n=1}^{\infty} |a_n|\), то сходится ряд \(\sum_{n=1}^{\infty} a_n\), и в этом случае ряд \(\sum_{n=1}^{\infty} a_n\) называется абсолютно сходящимся.
	\item Если расходится ряд \(\sum_{n=1}^{\infty} |a_n|\), но сходится ряд \(\sum_{n=1}^{\infty} a_n\), то в этом случае ряд \(\sum_{n=1}^{\infty} a_n\) называется условно сходящимся.
\end{itemize}
